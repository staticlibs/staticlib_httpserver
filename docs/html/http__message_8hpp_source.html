598  bool has_header_value(const std::string& key, const std::string& value) const {
599  auto range = m_headers.equal_range(key);
600  for (auto it = range.first; it != range.second; ++it) {
601  if (std::string::npos == it->second.find(',')) {
602  if (sl::utils::iequals(value, it->second)) {
603  return true;
604  }
605  } else {
606  auto vec = sl::utils::split(it->second, ',');
607  for (auto& el : vec) {
608  auto trimmed = sl::utils::trim(el);
609  if (sl::utils::iequals(value, trimmed)) {
610  return true;
611  }
612  }
613  }
614  }
615  return false;
616  }
