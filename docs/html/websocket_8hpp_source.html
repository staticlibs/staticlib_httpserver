513  static void send_internal(std::unique_ptr<websocket> self, Handler handler) {
514  auto self_ptr = self.get();
515  auto self_shared = sl::support::make_shared_with_release_deleter(self.release());
516  auto send_handler =
517  [self_shared, handler](const asio::error_code& ec, size_t) {
518  auto self = sl::support::make_unique_from_shared_with_release_deleter(self_shared);
519  if (nullptr != self.get()) {
520  self->clear_payload();
521  if (!ec) {
522  handler(std::move(self));
523  } else {
524  STATICLIB_PION_LOG_DEBUG("staticlib.pion.websocket", "Write error," <<
525  " code: [" << ec << "]" <<
526  " message: [" << ec.message() << "]" <<
527  " id: [" << self->get_id() << "]" <<
528  " path: [" << self->request->get_resource() << "]");
529  on_close(std::move(self), close_status::error);
530  }
531  } else {
532  STATICLIB_PION_LOG_WARN("staticlib.pion.websocket", "Lost context detected in 'async_write'");
533  }
534  };
535  auto send_handler_stranded = self_ptr->connection->get_strand().wrap(send_handler);
536  self_ptr->connection->async_write(self_ptr->payload_buffers, std::move(send_handler_stranded));

595  static void consume(std::unique_ptr<websocket> self, sl::io::span<const char> buf) {
596  // append incoming data to receive buffer
597  if (self->receive_buffer_overflow(buf.size())) {
598  STATICLIB_PION_LOG_WARN("staticlib.pion.websocket", "Receive buffer overflow" <<
599  " id: [" << self->get_id() << "]" <<
600  " path: [" << self->request->get_resource() << "]");
601  on_close(std::move(self), close_status::overflow);
602  return;

604  self->add_to_receive_buffer(buf);

608  static void process_receive_buffer(std::unique_ptr<websocket> self) {
609  auto frame = sl::websocket::frame(self->receive_buffer_span());
610  // close connection for invalid frame
611  if (!frame.is_well_formed()) {
612  STATICLIB_PION_LOG_WARN("staticlib.pion.websocket",
613  "Invalid frame received, header: [" << frame.header_hex() << "]" <<
614  " id: [" << self->get_id() << "]" <<
615  " path: [" << self->request->get_resource() << "]");
616  on_close(std::move(self), close_status::protocol_error);
617  return;
618  }
619  // read more data for fragmented frame
620  if (!frame.is_complete()) {
621  receive_internal(std::move(self));
622  return;
623  }
624  // frame is not fragmented on TCP, but may be partial
625  if (!frame.is_final()) {
626  on_continuation(std::move(self), frame);
627  return;
628  }
629  // complete and final frame received
630  process_final_frame(std::move(self), frame);
631  }

633  static void process_final_frame(std::unique_ptr<websocket> self, sl::websocket::frame frame) {
634  self->frames.push_back(frame);
635  // choose handler
636  switch (frame.type()) {
637  case sl::websocket::frame_type::text:
638  case sl::websocket::frame_type::binary:
639  case sl::websocket::frame_type::continuation:
640  on_message(std::move(self));
641  break;
642  case sl::websocket::frame_type::ping:
643  on_ping(std::move(self), frame);
644  break;
645  case sl::websocket::frame_type::close:
646  on_close(std::move(self), close_status::normal);
647  break;
648  default:
649  on_close(std::move(self), close_status::protocol_error);
650  break;
651  };
652  }

674  static void on_ping(std::unique_ptr<websocket> self, sl::websocket::frame frame) {
675  if (frame.payload_length() > 0) {
676  auto src = frame.payload_unmasked();
677  auto buf = std::array<char, (1<<7)>();
678  auto len = src.read({buf.data(), buf.size()});
679  self->write({buf.data(), len});
680  }

682  [](std::unique_ptr<websocket> self) {
683  receive(std::move(self));
684  });

687  static void on_continuation(std::unique_ptr<websocket> self, sl::websocket::frame frame) {
688  if (self->frames_cache_overflow()) {
689  STATICLIB_PION_LOG_DEBUG("staticlib.pion.websocket", "Frame fragments overflow" <<
690  " id: [" << self->get_id() << "]" <<
691  " path: [" << self->request->get_resource() << "]");
692  on_close(std::move(self), close_status::overflow);
693  return;
694  }
695  self->add_to_frames_cache(frame);
696  // safety precaution for stack overflow
697  post_with_strand(std::move(self), websocket::process_receive_buffer);
698  }
