118  void update_request_info(const http_request& http_request) {
119  request_method = http_request.get_method();
120  if (http_request.get_version_major() == 1 && http_request.get_version_minor() >= 1) {
121  set_chunks_supported(true);
122  } else if (http_request.get_version_major() == 0) {
123  // the request is likely HTTP 0.9 "simple-request", so expect the response to contain no header and no version info
124  set_status_code(0U);
125  set_status_message("");
126  set_version_major(0);
127  set_version_minor(0);
128  }
