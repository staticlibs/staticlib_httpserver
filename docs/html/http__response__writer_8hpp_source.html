310  static void handle_write(std::unique_ptr<http_response_writer> self,
311  const std::error_code& ec, std::size_t bytes_written) {
312  if (!ec) {
313  // response sent OK
314  if (self->sending_chunked_message()) {
315  STATICLIB_PION_LOG_DEBUG("staticlib.pion.http_response_writer",
316  "Sent HTTP response chunk of " << bytes_written << " bytes");

318  STATICLIB_PION_LOG_DEBUG("staticlib.pion.http_response_writer",

332  template <typename SendHandler>
333  void send_more_data(const bool send_final_chunk, SendHandler send_handler) {
334  // make sure that we did not lose the TCP connection
335  if (tcp_conn->is_open()) {
336  // make sure that the content-length is up-to-date
337  // prepare the write buffers to be sent
338  auto write_buffers = std::vector<asio::const_buffer>();
339  prepare_write_buffers(write_buffers, send_final_chunk);
340  // send data in the write buffers
341  auto send_handler_stranded = tcp_conn->get_strand().wrap(send_handler);
342  tcp_conn->async_write(write_buffers, send_handler_stranded);
343  } else {
344  tcp_conn->finish();
345  }

354  void prepare_write_buffers(std::vector<asio::const_buffer>& write_buffers,
355  const bool send_final_chunk) {
356  // check if the HTTP headers have been sent yet
357  if (! sent_headers) {
358  // initialize write buffers for send operation
359  prepare_buffers_for_send(write_buffers);
360 
361  // only send the headers once
362  sent_headers = true;
363  }
364 
365  // combine I/O write buffers (headers and content) so that everything
366  // can be sent together; otherwise, we would have to send headers
367  // and content separately, which would not be as efficient
368 
369  // don't send anything if there is no data in content buffers
370  if (content_length > 0) {
371  if (supports_chunked_messages() && sending_chunked_message()) {
372  // prepare the next chunk of data to send
373  // write chunk length in hex
374  auto cast_buf = std::array<char, 35>();
375  #ifdef _WIN32
376  auto written = sprintf_s(cast_buf.data(), cast_buf.size(), "%lx", static_cast<long>(content_length));
377  #else // !WIN32
378  auto written = sprintf(cast_buf.data(), "%lx", static_cast<long>(content_length));
379  #endif // WIN32
380 
381  // add chunk length as a string at the back of the text cache
382  // append length of chunk to write_buffers
383  write_buffers.push_back(add_to_cache({cast_buf.data(), written}));
384  // append an extra CRLF for chunk formatting
385  write_buffers.push_back(asio::buffer(http_message::STRING_CRLF));
386 
387  // append response content buffers
388  write_buffers.insert(write_buffers.end(), content_buffers.begin(),
389  content_buffers.end());
390  // append an extra CRLF for chunk formatting
391  write_buffers.push_back(asio::buffer(http_message::STRING_CRLF));
392  } else {
393  // append response content buffers
394  write_buffers.insert(write_buffers.end(), content_buffers.begin(),
395  content_buffers.end());
396  }
397  }
398 
399  // prepare a zero-byte (final) chunk
400  if (send_final_chunk && supports_chunked_messages() && sending_chunked_message()) {
401  // add chunk length as a string at the back of the text cache
402  char zero = '0';
403  // append length of chunk to write_buffers
404  write_buffers.push_back(add_to_cache({std::addressof(zero), 1}));
405  // append an extra CRLF for chunk formatting
406  write_buffers.push_back(asio::buffer(http_message::STRING_CRLF));
407  write_buffers.push_back(asio::buffer(http_message::STRING_CRLF));
408  }
409  }

417  asio::const_buffer add_to_cache(sl::io::span<const char> data) {
418  payload_cache.emplace_back(new char[data.size()]);
419  char* dest = payload_cache.back().get();
420  std::memcpy(dest, data.data(), data.size());
421  return asio::buffer(dest, data.size());
